def HashtoXML(hash, parent = nil)
  if parent.nil?
    parent = Ox::Document.new(:version => '1.0')
  end

  hash.each do |key, value|
    if value.is_a?(Hash)
      key = Ox::Element.new(key)
      parent << key
      self.HashtoXML(value, key)
    else
      key = Ox::Element.new(key)
      parent << key
      key << value
    end
  end
  
  return Ox.dump(parent)
end
