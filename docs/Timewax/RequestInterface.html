def post(data, domain, path) 
    uri = URI("https://#{domain}/#{path}")

    http = Net::HTTP.new(uri.hostname)
    req = Net::HTTP::Post.new(path)
    req.content_type = 'text/xml'
    req.body = data

    # puts "body request: #{data}"
    # puts "- - - - - - - - - - - -"
    
    return http.request(req)
end
